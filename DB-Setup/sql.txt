CREATE DATABASE webscp_appointments;
CREATE TABLE `webscp_appointments`.`appointments` ( `A-ID` MEDIUMINT UNSIGNED NOT NULL , `Titel` VARCHAR(50) NULL DEFAULT NULL , `Text` VARCHAR(1000) NULL DEFAULT NULL , `Place` VARCHAR(200) NULL DEFAULT NULL , `Date` DATE NULL DEFAULT NULL , `Expiration` DATE NULL DEFAULT NULL , PRIMARY KEY (`A-ID`)) ENGINE = InnoDB; 
CREATE TABLE `webscp_appointments`.`options` ( `A-ID` MEDIUMINT UNSIGNED NOT NULL , `O-ID` MEDIUMINT UNSIGNED NOT NULL , `Titel` VARCHAR(50) NULL DEFAULT NULL , `Date` DATE NULL DEFAULT NULL , PRIMARY KEY (`O-ID`), INDEX `A-ID` (`A-ID`)) ENGINE = InnoDB; 
CREATE TABLE `webscp_appointments`.`user-input` ( `O-ID` MEDIUMINT UNSIGNED NOT NULL , `Username` VARCHAR(30) NOT NULL , `Comment` VARCHAR(500) NULL DEFAULT NULL , INDEX `O-ID` (`O-ID`)) ENGINE = InnoDB;
