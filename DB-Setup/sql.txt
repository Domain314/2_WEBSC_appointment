CREATE DATABASE webscp_appointments;
CREATE TABLE `webscp_appointments`.`appointments` ( `aID` MEDIUMINT UNSIGNED NOT NULL , `Titel` VARCHAR(50) NULL DEFAULT NULL , `Text` VARCHAR(1000) NULL DEFAULT NULL , `Place` VARCHAR(200) NULL DEFAULT NULL , `Date` DATETIME NULL DEFAULT NULL , `Expiration` DATETIME NULL DEFAULT NULL , PRIMARY KEY (`aID`)) ENGINE = InnoDB;
CREATE TABLE `webscp_appointments`.`options` ( `aID` MEDIUMINT UNSIGNED NOT NULL , `oID` MEDIUMINT UNSIGNED NOT NULL , `Titel` VARCHAR(50) NULL DEFAULT NULL , `Date` DATETIME NULL DEFAULT NULL , PRIMARY KEY (`oID`), INDEX `aID` (`aID`)) ENGINE = InnoDB;
CREATE TABLE `webscp_appointments`.`userinput` ( `oID` MEDIUMINT UNSIGNED NOT NULL , `Username` VARCHAR(30) NOT NULL , `Comment` VARCHAR(500) NULL DEFAULT NULL , INDEX `oID` (`oID`)) ENGINE = InnoDB;
